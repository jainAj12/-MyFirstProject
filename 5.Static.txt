class JavaExample2{
   static int num;
   static String mystr;
   //First Static block
   static{
      System.out.println("Static Block 1");
      num = 68;
      mystr = "Block1";
  } 
  //Second static block
  static{
      System.out.println("Static Block 2");
      num = 98;
      mystr = "Block2";
  }
  public static void main(String args[])
  {
      System.out.println("Value of num: "+num);
      System.out.println("Value of mystr: "+mystr);
   }
}

public class Counter {

    int myCount = 0;

    void increment ( ) {
        myCount = myCount + 1;
    }

    void print ( ) {
        System.out.println ("count = " + myCount);
    }

    public static void main (String [ ] args) {
        Counter c1 = new Counter ( );
        c1.increment ( );  // c1's myCount is now 1
        c1.increment ( );  // c1's myCount is now 2
        Counter c2 = new Counter ( );
        c2.increment ( );  // c2's myCount is now 1
        c1.increment ( );  // c1's myCount is now 3
        c1.myCount = 0;    // effectively resets the c1 counter
    }
}
